```
/**
 * Exclusive-or gate:
 * out = not (a == b)
 */

CHIP Xor {
    IN a, b;
    OUT out;

    PARTS:
    // Put your code here:
    Not(in=a,out=na);
	Not(in=b,out=nb);
	And(a=na,b=b,out=o1);
	And(a=a,b=nb,out=o2);
	Or(a=o1,b=o2,out=out);
}
```
